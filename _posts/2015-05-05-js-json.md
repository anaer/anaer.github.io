#### js 动态生成json对象、时时更新json对象 

```js

var str1 = {"name": "apple", "sex": "21"};
                 
// 参数：prop = 属性，val = 值
function createJson(prop, val) {
    // 如果 val 被忽略
    if(typeof val === "undefined") {
        // 删除属性
        delete str1[prop];
    }
    else {
        // 添加 或 修改
        str1[prop] = val;
    }
}
```
